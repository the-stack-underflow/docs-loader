Utf8View iterates the code points of a utf-8 encoded string. 

var utf8 = (try std.unicode.Utf8View.init("hi there")).iterator();
while (utf8.nextCodepointSlice()) |codepoint| {
  std.debug.print("got codepoint {}\n", .{codepoint});
}
